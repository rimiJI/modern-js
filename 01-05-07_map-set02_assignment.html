🚩map-set assignment

각 문법에 따른 코드를 빠르게 머릿속으로 구현하시오
방법1.(Array.from: ) 

    function unique(arr) {
      /* 제출 답안 */
      return Array.from(new Set(arr));
    }

Array.from()은 정적메서드로 순회기능or유사배열 객체에서 얕게 복사된 새로운 Array 인스턴스를 생성합니다.

방법2.(for...of) 

    function uniqueLoop(arr) {
      let result = []; // 결과를 담을 빈 배열

      for (let str of arr) {
        // 배열을 순회하면서
        if (!result.includes(str)) {
          // 결과 배열에 현재 요소가 없다면
          result.push(str); // 추가한다
        }
      }
      return result;
    }
방법3.(스프레드문법[...]) 

    function uniqueSpread(arr) {
      return [...new Set(arr)];
    }

스프레드문법은 짧고 세련된 최신문법임. ...로 펼치고, [ ]로 묶어주는 방법
